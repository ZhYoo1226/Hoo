class SkyQueryCallbacks(QueryCallbacks):
    """A no-op implementation of QueryCallbacks."""

    def on_context(self, context: Any) -> None:
        """Handle when context data is constructed."""

    def on_map_response_start(self, map_response_contexts: list[str]) -> None:
        """Handle the start of map operation."""

    def on_map_response_end(self, map_response_outputs: list[SearchResult]) -> None:
        """Handle the end of map operation."""

    def on_reduce_response_start(
            self, reduce_response_context: str | dict[str, Any]
    ) -> None:
        """Handle the start of reduce operation."""

    def on_reduce_response_end(self, reduce_response_output: str) -> None:
        """Handle the end of reduce operation."""

    # 执行 shell 命令的通用方法
    def run_command(self, cmd: List[str]) -> str:
        try:
            print(f"execute cmd : {cmd}")
            result = subprocess.run(cmd, capture_output=True, text=True, check=True)
            return result.stdout.strip()
        except subprocess.CalledProcessError as e:
            raise Exception(f"Command execution failed: {e.stderr}")
        except Exception as e:
            raise Exception(f"Unexpected error: {str(e)}")

    # 执行异步shell命令
    def async_run_command(self, cmd: List[str]) -> str:
        try:
            print(f"execute cmd : {cmd}")
            process = subprocess.Popen(cmd, stdout=sys.stdout, stderr=sys.stderr, text=True)

            def monitor():
                process.wait()
                if process.returncode == 0:
                    self.update_index_state("FINISH")
                    time.sleep(3)
                    self.update_index_state(None)

                else:
                    self.update_index_state("ERROR")

            threading.Thread(target=monitor, daemon=True).start()
            # stdout, stderr = process.communicate()
            # if process.returncode != 0:
            #     raise RuntimeError(f"Command failed with code {process.returncode}: {stderr}")
            # return stdout.strip()
            return "success"
        except subprocess.CalledProcessError as e:
            self.update_index_state("ERROR")
            raise Exception(f"Command execution failed: {e.stderr}")
        except Exception as e:
            self.update_index_state("ERROR")
            raise Exception(f"Unexpected error: {str(e)}")

@app.post("/graph/store", response_model=Response, summary="上传文件或者知识图谱文本,不能同时上传文件file和文本text")
async def upload_file(
        subject: str = Form(default=None, description="分类主题"),
        file: UploadFile = File(default=None, description="上传的文件（doc, docx, pdf 等）"),
        text: str = Form(default=None, description="纯文本内容"),
        fileKey: str = Form(default=None, description="文件或文本的唯一标识（可选）")
):
    try:
        if file and text:
            raise ValueError("不能同时上传文件和文本")
        if subject in g_graph_servers:
            graph = g_graph_servers[subject]
        else:
            graph = GraphServer(subject)
            g_graph_servers[subject] = graph
        output = {}
        if file:
            # TODO 需要根据doc，pdf进行转换处理
            first_chunk = await file.read(8192)
            if not first_chunk:
                raise ValueError("上传的文件为空")

            size = len(first_chunk)
            await file.seek(0)

            file_extension = os.path.splitext(file.filename)[-1]
            print(f"upload file:{file.filename},ext={file_extension}")
            match file_extension:
                case ".txt":
                    output = graph.add_docs(str(await file.read(), encoding='utf-8'), fileKey=fileKey)
                case ".doc":
                    # raise ValueError("不支持的文件格式")
                    try:
                        output = await process_doc_file(file, file_extension)
                    except Exception as doc_err:
                        print(f"[WARN] antiword 处理失败: {doc_err}")
                        await file.seek(0)
                        output = await process_docx_zipfile(file)
                    output = graph.add_docs(output, fileKey=fileKey)
                case ".docx":
                    # raise ValueError("不支持的文件格式")
                    if size < 50 * 1024 * 1024:
                        output = await process_docx_in_memory(file)
                    else:
                        output = await process_docx_with_tempfile(file, file_extension)
                    output = graph.add_docs(output, fileKey=fileKey)
                case ".pdf":
                    # raise ValueError("不支持的文件格式")
                    if size < 50 * 1024 * 1024:
                        output = await process_pdf_in_memory(file)
                    else:
                        output = await process_pdf_with_tempfile(file, file_extension)
                    output = graph.add_docs(output, fileKey=fileKey)
                case ".xls":
                    raise ValueError("不支持的文件格式")
        elif text:
            output = graph.add_docs(text, fileKey=fileKey)
        else:
            raise ValueError("未提供文件或文本")
        return Response(code=0, msg="success", data=output)

    except Exception as e:
        return Response(code=-1, msg=str(e))

                            print(f"[DEBUG] 开始处理 docx: {file.filename}")
                            try:
                                output = await process_docx_in_memory(file)
                                print(f"[DEBUG] docx 转换完成, output 长度: {len(output) if output else 0}")
                            except Exception as docx_err:
                                print(f"[WARN] docx 转换失败: {docx_err}, 尝试用 zipfile 方式处理")
                                import traceback
                                traceback.print_exc()
                                # 回退前，先 seek 回文件开头
                                await file.seek(0)
                                try:
                                    output = await process_docx_zipfile(file)
                                    print(f"[DEBUG] zipfile 转换完成")
                                except Exception as zip_err:
                                    print(f"[WARN] zipfile 也失败: {zip_err}, 尝试用 antiword 处理")
                                    await file.seek(0)
                                    output = await process_doc_file(file, file_extension)
                                    print(f"[DEBUG] antiword 转换完成")
                        else:
                            output = await process_docx_with_tempfile(file, file_extension)
                        print(f"[DEBUG] 调用 graph.add_docs, fileKey: {fileKey}")
                        output = graph.add_docs(output, fileKey=fileKey)
                        print(f"[DEBUG] add_docs 完成")
                    case ".pdf":
                        # raise ValueError("不支持的文件格式")
                        if size < 50 * 1024 * 1024:
                            output = await process_pdf_in_memory(file)
                        else:
                            output = await process_pdf_with_tempfile(file, file_extension)
                        output = graph.add_docs(output, fileKey=fileKey)
                    case ".xls":
                        raise ValueError("不支持的文件格式")
                results.append({"filename": file.filename, "result": output})
            except Exception as e:
                return Response(code=-1, msg=str(e))
        else:
            raise ValueError("未提供文件")
    return Response(code=0, msg="success", data={'result': results})

@app.get("/api/status/stream/{subject}", summary="SSE 推送索引状态")
async def graph_status_sse(
        subject: str
):
    if subject not in g_graph_servers:
        # 主题不存在，直接返回 404
        return StreamingResponse(
            (f"data: {json.dumps({'ready': False, 'message': '主题不存在'}, ensure_ascii=False)}\n\n" for _ in
             range(1)),
            media_type="text/event-stream"
        )
    graph = g_graph_servers[subject]

    async def event_generator():
        # out_dir = Path(BASE_DIR) / "datas" / subject / "output"
        # graph = g_graph_servers[subject]
        # print(out_dir)
        while True:
            state = graph.index_state
            # 用 match-case（Python 3.10+）最直观；低版本可改用 if-elif
            match state:
                case "PROCESSING":
                    msg = "正在构建知识图谱向量库"
                    graph._error_reset_done = False  # 重新开始时重置标记
                    # scheduler_logger.info(f"[SSE][{subject}] {msg}")
                    yield f"data: {json.dumps({'ready': False, 'message': msg, 'state': 'PROCESSING'}, ensure_ascii=False)}\n\n"
                    await asyncio.sleep(2)
                case "ERROR":
                    # reset_snapshot()
                    error_date = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
                    scheduler_logger.warning(f"构建知识图谱向量库出现错误,请重新尝试,错误时间:{error_date}")
                    if graph.index_state == "ERROR" and not graph._error_reset_done:
                        reset_snapshot()
                        scheduler_logger.warning(f"[{subject}] 快照已重置，避免重复清空")
                        graph._error_reset_done = True
                        # graph.update_index_state(None)  # ✅ 关键：清空后立刻重置状态
                    msg = f"构建知识图谱向量库出现错误,请重新尝试,错误时间:{error_date}"
                    yield f"data: {json.dumps({'ready': False, 'message': msg, 'state': 'ERROR'}, ensure_ascii=False)}\n\n"
                    await asyncio.sleep(2)
                    # msg = "构建知识图谱向量库出现错误,请重新尝试"
                    # # scheduler_logger.error(f"[SSE][{subject}] {msg}")
                    # yield f"data: {json.dumps({'ready': False, 'message': msg,'state':'ERROR'},ensure_ascii=False)}\n\n"
                    # await asyncio.sleep(2)
                    # break
                case "FINISH":
                    msg = "构建知识图谱向量库完成!"
                    finish_date = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
                    # scheduler_logger.info(f"[SSE][{subject}] {msg}")
                    scheduler_logger.warning(f"构建知识图谱向量库完成!完成时间:{finish_date}")
                    yield f"data: {json.dumps({'ready': True, 'message': msg, 'state': 'FINISH'}, ensure_ascii=False)}\n\n"
                    await asyncio.sleep(2)
                    graph.update_index_state(None)
                    # break  # 结束 SSE
                case None:
                    msg = "暂无知识图谱向量库构建请求"
                    # scheduler_logger.debug(f"[SSE][{subject}] {msg}")
                    yield f"data: {json.dumps({'ready': False, 'message': msg, 'state': 'NONE'}, ensure_ascii=False)}\n\n"
                    await asyncio.sleep(2)
                case _:
                    # 未知状态，也按 None 处理（或自定义日志）
                    msg = f"未知状态: {state}"
                    # scheduler_logger.warning(f"[SSE][{subject}] {msg}")
                    yield f"data: {json.dumps({'ready': False, 'message': msg, 'state': 'UNKNOWN'}, ensure_ascii=False)}\n\n"
                    await asyncio.sleep(2)

    return StreamingResponse(
        event_generator(),
        media_type="text/event-stream"
    )

import httpx
from pathlib import Path


@app.get("/graph/auto/test/{subject}", summary="TCS 测试图谱自动化流程接口（批量上传）")
async def auto_test(subject: str):
    """
    一键完成：
    1. /graph/init
    2. 把 test_auto_upload/ 下所有文件一次性 POST 到 /graph/batch_store
    3. /graph/index
    4. 更新本地快照
    """
    # base_url = "http://127.0.0.1:5000"
    # mount_dir = Path(BASE_DIR) / "datas" / "test" / "input"
    file_dir = Path(BASE_DIR) / "input"
    # if not mount_dir.is_dir():
    #     return {"code": 1, "msg": f"{mount_dir} 不存在，请先放入测试文件"}
    if not file_dir.is_dir():
        return {"code": 1, "msg": f"{file_dir} 不存在，请先放入待上传文件"}

    # 1. 差异检测
    current = scan2dict(str(file_dir))
    last = load_snapshot()
    graph = g_graph_servers[subject]
    if graph.index_state == "PROCESSING":
        scheduler_logger.info("  - 索引任务正在运行中!")
        return {"code": -1, "msg": "索引任务正在运行中，请稍后再试"}

    if current == last:
        return {"code": -1, "msg": "无差异，无需更新"}

    try:
        async with httpx.AsyncClient(timeout=None) as client:
            # 2. 初始化（如需）
            # await client.post(f"{BASE_URL}/graph/init", data={"subject": subject})
            # 更新快照
            save_snapshot(current)
            scheduler_logger.info("  - 发现差异， 已更新比对表（快照）")
            # 3批量删除之前并备份之前内容
            scheduler_logger.info(" - 准备备份当前向量库并删除当前向量库 /graph/batch_delete/store/")
            await client.delete(f"{BASE_URL}/graph/batch_delete/store/{subject}")
            print('之前内容已全部清除')
            # 4. 批量上传 file_dir 下的文件到 /graph/tcs/batch_store
            upload_files = []
            file_keys = []
            scheduler_logger.info(f"  - 开始收集上传文件，目录: {file_dir}")
            for p in file_dir.glob("*"):
                if not p.is_file():
                    continue
                ext = p.suffix.lower()
                if ext not in allowed:
                    # 可选：print(f"skip {p.name} (ext={ext})")
                    continue
                file_bytes = p.read_bytes()  # 预读为 bytes，避免 AsyncClient 抱怨同步流
                scheduler_logger.info(f"    · 将上传文件: {p.name}, size={len(file_bytes)} bytes, ext={ext}")
                upload_files.append(
                    ("files", (p.name, file_bytes, "application/octet-stream"))
                )
                file_keys.append(p.stem)
            if not upload_files:  # 没有文件可传
                return {"code": 1, "msg": f"{file_dir} 下未找到可上传的文件"}
            # FastAPI 接口要求 subject 在表单字段中，files 单独放在 files 中
            # 注意：需要把 fileKeys 作为一个列表传递，requests 合并同名参数时会用逗号分隔
            # 所以需要在接收端 split 处理
            form_data = [("subject", subject)]
            form_data.append(("fileKeys", ",".join(file_keys)))  # 用逗号分隔合并
            scheduler_logger.info(f"  - form_data={form_data}")
            scheduler_logger.info("  - 正在调用 /graph/tcs/batch_store 上传文件")

            # 使用同步 requests 发送 multipart，并放到线程池执行，彻底避开 AsyncClient 的同步流限制
            def _sync_upload():
                return requests.post(
                    f"{BASE_URL}/graph/tcs/batch_store",
                    data=form_data,
                    files=upload_files,
                    timeout=None
                )

            try:
                resp = await asyncio.to_thread(_sync_upload)
                resp.raise_for_status()
            except Exception as exc:
                scheduler_logger.exception(
                    f"  - 调用 /graph/tcs/batch_store 失败: {type(exc)} {exc}"
                )
                raise
            # 4. 构建索引
            scheduler_logger.info("  - 发现差异，准备调用 /graph/index")
            await client.post(
                f"{BASE_URL}/graph/index",
                data={"subject": subject, "method": "standard"}
            )
            # # 5. SSE记录任务状态
            # await client.get(f"{BASE_URL}/graph/status/stream/{subject}")
        return {"code": 0, "msg": "开始更新图谱向量库", "subject": subject}
    except Exception as e:
        reset_snapshot()
        scheduler_logger.info(f"请求更新向量库失败,失败原因{e}")
        return {"code": -1, "msg": str(e)}

async def verify_llm_health(auth_header: str) -> tuple[bool, str]:
    """
    Quick health check against the configured LLM endpoint.
    Returns (True, None) if the check passes or (False, reason) otherwise.
    """
    if not LLM_HEALTHCHECK_URL:
        return True, None
    headers = {"Content-Type": "application/json"}
    api_key = LLM_HEALTHCHECK_API_KEY
    if not api_key and auth_header and auth_header.startswith("Bearer "):
        api_key = auth_header.split(" ", 1)[1]
    if api_key:
        headers["Authorization"] = f"Bearer {api_key}"
    payload = {
        "model": LLM_HEALTHCHECK_MODEL,
        "messages": [
            {"role": "system", "content": "connection test"},
            {"role": "user", "content": "ping"},
        ],
        "stream": False
    }
    try:
        async with httpx.AsyncClient(timeout=LLM_HEALTHCHECK_TIMEOUT) as client:
            resp = await client.post(LLM_HEALTHCHECK_URL, json=payload, headers=headers)
            resp.raise_for_status()
            data = resp.json()
            if "choices" in data:
                return True, None
            return False, f"unexpected response: {data}"
    except Exception as exc:
        return False, str(exc)

def run_web_server():
    parser = argparse.ArgumentParser()
    parser.add_argument("--host", type=str, default="0.0.0.0", required=False, help="host ip, localhost, 0.0.0.0")
    parser.add_argument("--port", type=int, default=5000, required=False, help="server port")
    args = parser.parse_args()
    app.add_middleware(
        CORSMiddleware,
        # 允许跨域的源列表，例如 ["http://www.example.org"] 等等，["*"] 表示允许任何源
        allow_origins=["*"],
        # 跨域请求是否支持 cookie，默认是 False，如果为 True，allow_origins 必须为具体的源，不可以是 ["*"]
        allow_credentials=False,
        # 允许跨域请求的 HTTP 方法列表，默认是 ["GET"]
        allow_methods=["*"],
        # 允许跨域请求的 HTTP 请求头列表，默认是 []，可以使用 ["*"] 表示允许所有的请求头
        # 当然 Accept、Accept-Language、Content-Language 以及 Content-Type 总之被允许的
        allow_headers=["*"],
        # 可以被浏览器访问的响应头, 默认是 []，一般很少指定
        # expose_headers=["*"]
        # 设定浏览器缓存 CORS 响应的最长时间，单位是秒。默认为 600，一般也很少指定
        # max_age=1000
    )
    uvicorn.run("web_app:app",
                host='0.0.0.0',
                port=args.port,
                reload=False,
                # workers=1,
                log_level="debug",
                timeout_keep_alive=5)
